// ═══════════════════════════════════════════════ // 2liv — Apps Script // Cole este código no Apps Script da sua planilha // ═══════════════════════════════════════════════ const PASTA_DRIVE_ID = '1M6HZnGKYafqXjcSW67ZRY23BkseARCcT'; const ABA_NOME = 'Ordens'; // nome da aba na planilha const COLUNAS = [ 'Mês/Ano', 'Data/Hora Solicitação', 'Dias p/ Vencer', 'Status', 'Prioridade', 'Solicitante', 'E-mail', 'Evento', 'Data Evento', 'Fornecedor', 'CPF/CNPJ', 'Titular', 'PIX', 'Banco', 'Tipo Recebimento', 'Link Boleto', 'Forma Pagamento', 'Conta Pagamento', 'Centro de Custo', 'Tipo Documento', 'Link NF/Recibo', 'Condição', 'Total Ordem', 'Nº Parcelas', 'Intervalo', 'Parcelas (detalhes)', 'Descrição Item', 'Valor Item', 'Vencimento Item', 'Observações', // Colunas do financeiro (preenchidas manualmente na planilha) 'Aprovado por', 'Motivo Recusa', 'NF/Recibo Confirmado', 'Contrato no Drive', 'Apto p/ Pagamento', 'Comprovante', 'Data Pagamento', ]; // ── Ponto de entrada ────────────────────────── function doPost(e) { try { const data = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.getActiveSpreadsheet(); const aba = getOrCreateAba(ss); // Garante cabeçalho if (aba.getLastRow() === 0) { aba.appendRow(COLUNAS); aba.getRange(1, 1, 1, COLUNAS.length) .setBackground('#1B4FD8') .setFontColor('#ffffff') .setFontWeight('bold'); aba.setFrozenRows(1); } const pasta = DriveApp.getFolderById(PASTA_DRIVE_ID); // Upload do boleto (se houver) let linkBoleto = ''; if (data.fileBoletoBase64 && data.nomeArquivoBoleto) { linkBoleto = uploadArquivo(pasta, data.nomeArquivoBoleto, data.fileBoletoBase64, data.fileBoletoMime || 'application/octet-stream'); } // Upload da NF ou Recibo let linkDoc = ''; if (data.fileDocBase64 && data.nomeArquivoDoc) { linkDoc = uploadArquivo(pasta, data.nomeArquivoDoc, data.fileDocBase64, data.fileDocMime || 'application/octet-stream'); } // Parcelas resumidas const parcelasStr = data.parcelas && data.parcelas.length ? data.parcelas.map(p => 'P' + p.numero + ': R$ ' + p.valor + ' (' + p.vencimento + ')').join(' | ') : ''; // Uma linha por item de pagamento const items = data.items || [{}]; items.forEach(function(item) { const linha = [ data.mesAno || '', data.dataHoraSolicitacao || new Date().toLocaleString('pt-BR'), item.diasParaVencer || '', 'Pendente', data.prioridade || '', data.solicitanteNome || '', data.solicitanteEmail || '', data.evento || '', data.dataEvento || '', data.fornecedor || '', data.documento || '', data.titular || '', data.pix || '', data.banco || '', data.tipoRecebimento || '', linkBoleto, data.formaPagamento || '', data.contaPagamento || '', data.centroCusto || '', data.tipoDocumento || '', linkDoc, data.condicao || '', data.totalOrdem || '', data.nParcelas || '', data.intervalo || '', parcelasStr, item.descricao || '', item.valor || '', item.vencimento || '', data.observacoes || '', // Financeiro — vazios, preenchidos na planilha '', '', '', '', '', '', '', ]; aba.appendRow(linha); }); // Formata a planilha formatarAba(aba); return ContentService .createTextOutput(JSON.stringify({ ok: true })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService .createTextOutput(JSON.stringify({ ok: false, erro: err.message })) .setMimeType(ContentService.MimeType.JSON); } } // ── Helpers ─────────────────────────────────── function getOrCreateAba(ss) { return ss.getSheetByName(ABA_NOME) || ss.insertSheet(ABA_NOME); } function uploadArquivo(pasta, nome, base64, mime) { try { const bytes = Utilities.base64Decode(base64); const blob = Utilities.newBlob(bytes, mime, nome); const file = pasta.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); return file.getUrl(); } catch (e) { return 'Erro no upload: ' + e.message; } } function formatarAba(aba) { const ultima = aba.getLastRow(); if (ultima < 2) return; // Larguras das colunas aba.setColumnWidth(1, 80); // Mês/Ano aba.setColumnWidth(2, 160); // Data/Hora aba.setColumnWidth(4, 100); // Status aba.setColumnWidth(5, 110); // Prioridade aba.setColumnWidth(8, 180); // Evento aba.setColumnWidth(10, 180); // Fornecedor aba.setColumnWidth(27, 220); // Descrição Item // Cor de fundo por status (coluna 4) const statusRange = aba.getRange(2, 4, ultima - 1, 1); const valores = statusRange.getValues(); valores.forEach(function(row, i) { var cor = '#ffffff'; if (row[0] === 'Pendente') cor = '#fff8e1'; if (row[0] === 'Aprovado') cor = '#e8f5e9'; if (row[0] === 'Recusado') cor = '#ffebee'; if (row[0] === 'Pago') cor = '#e3f2fd'; aba.getRange(i + 2, 1, 1, COLUNAS.length).setBackground(cor); }); // Bordas aba.getRange(1, 1, ultima, COLUNAS.length) .setBorder(true, true, true, true, true, true); } // ── GET para teste ──────────────────────────── function doGet() { return ContentService .createTextOutput(JSON.stringify({ ok: true, msg: '2liv Apps Script funcionando!' })) .setMimeType(ContentService.MimeType.JSON); }
